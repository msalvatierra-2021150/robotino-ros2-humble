Source the ROS:
source /opt/ros/humble/setup.bash

Source the workspace:
source ~/robotino3/robotino3ros2/install/setup.bash

Connect with the robotino:
ros2 launch rto_bringup rto_bringup_launch.py hostname:=172.27.1.1

Move the robotino:
timeout 10s ros2 topic pub /rto3/cmd_vel geometry_msgs/msg/Twist "{linear: {x: 0.05, y: 0.0, z: 0.0}, angular: {x: 0.0, y: 0.0, z: 0.0}}" -r 1

Launch obstacle avoidance, RViz, and the URDF/XACRO robot description publisher.
ros2 launch rto_bringup rto_odom_sensors_rviz.launch.py hostname:=172.27.1.1

Publish the to /cmd_vel
ros2 topic pub /cmd_vel geometry_msgs/msg/Twist \
"{linear: {x: 0.2}, angular: {z: 0.0}}"

Generate the URDF
ros2 run xacro xacro rto_description/urdf/robots/rto-3.urdf.xacro > /tmp/rto-3.urdf

Publish the URDF to Gazebo
ros2 run gazebo_ros spawn_entity.py -entity robotino -file /tmp/rto-3.urdf

Launch gazebo
ros2 launch rto_simulation gazebo_model.launch.py

Generate MAP from slam_toolbox
ros2 run nav2_map_server map_saver_cli -f ~/robotino3/maps/robotino_map

Run nav2 with config yaml
ros2 launch nav2_bringup navigation_launch.py \
  use_sim_time:=true \
  params_file:=/home/mike/robotino3/robotino3ros2/rto_simulation/config/nav2_params_robotino.yaml

Run Frontier navigation
ros2 launch frontier_exploration classical_exploration.launch.py

Specify the frontier to explore to nav2
ros2 service call /frontier_pose frontier_interfaces/srv/FrontierGoal "{goal_rank: 0}"

Connect to via SSH to thw Robotino
ssh robotino@192.168.0.20

Connect to the Robot and start mapping
ros2 launch rto_bringup real_exploration.launch.py start_frontier:=false autostart:=false robot_ip:=192.168.0.20


Correct time in the Pi
./sync_pi_time_and_lidar.sh
